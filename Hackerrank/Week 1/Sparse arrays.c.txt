#include <stdio.h>
#include <string.h>
#define MAX 1000
#define LEN 21
int main(){
    int n,q;
    char strings[MAX][LEN];
    char queries[MAX][LEN];
    int result[MAX] = {0};
    scanf("%d", &n);
    for (int i =0; i<n; i++){
        scanf("%s", strings[i]);
        
    }
    scanf("%d", &q);
    for (int i=0; i<q; i++){
        scanf("%s", queries[i]);
    }
    for (int i=0; i<q; i++){
        for (int j=0; j<n; j++) {
            if (strcmp(queries[i], strings[j])==0){
                result[i]++;
            }
        }
    }
    for (int i=0; i<q; i++){
        printf("%d\n", result[i]);
    }
    return 0;
}
