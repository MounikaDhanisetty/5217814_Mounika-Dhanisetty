#include <stdio.h>
#include <ctype.h>
#include <string.h>

char* pangrams(char* s) {
    int alphabet[26] = {0}; 
    int index;

    for (int i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {
            index = tolower(s[i]) - 'a';
            alphabet[index] = 1;
        }
    }

    
    for (int i = 0; i < 26; i++) {
        if (alphabet[i] == 0) {
            return "not pangram";
        }
    }

    return "pangram";
}

int main() {
    char s[1001];  
    fgets(s, sizeof(s), stdin);  

    
    s[strcspn(s, "\n")] = '\0';

    printf("%s\n", pangrams(s));
    return 0;
}
