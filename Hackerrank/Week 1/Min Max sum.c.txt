#include <stdio.h>
#include <limits.h>
void minMaxSum(int arr[], int n){
    long long totalSum = 0;
    int min = INT_MAX;
    int max = INT_MIN;
    for (int i =0; i<n; i++){
        totalSum +=arr[i];
        if (arr[i] < min)
            min = arr[i];
        if (arr[i] > max)
            max = arr[i];
    }
    printf("%lld %lld\n", totalSum - max, totalSum - min);
    
}
int main(){
    int arr[5];
    for (int i = 0; i<5; i++){
        scanf("%d", &arr[i]);
    }
    minMaxSum(arr,5);
    return 0;
}
